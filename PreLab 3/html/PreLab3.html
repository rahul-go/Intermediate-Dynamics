% Given Values
k = 1;                          % Spring constant (N/m)
b = 1;                          % Drag constant (N*s/m)
m = 1;                          % Mass (N*s^2/m) aka (kg)

% Matrices A, B, C, D
A = [0, 1;
     -k/m, -b/m];
B = [0;
     1/m];
C = [1, 0;
     0, 1;
     -k/m, -b/m;
     k, 0];
D = [0;
     0;
     1/m;
     0];

% State Space System Object
sys = ss(A, B, C, D);
